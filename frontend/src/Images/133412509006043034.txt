namespace EventHandling2
{
    public class Demo
    {
        //Declare the delegate
         public delegate void MyDelegate();

        //Declare the Event
         public event MyDelegate MySampleEvent;

        //Create a Method which raises the Event
        public void RaiseMyEvent()
        {
            //Raise the Event
            MySampleEvent?.Invoke();
        }
    }

    public class Test
    {
        
	public void onMyEventRaised1()
        {
            //do some work event get raised
            Console.WriteLine("onMyEventRaise1");
        }
    }

   public class Test2
    {

	public void onMyEventRaised2()
        {
            //do some work event get raised
            Console.WriteLine("onMyEventRaise2");
        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            Demo obj = new Demo();
            Test objTest = new Test();
            Test2 objTest2 = new Test2();

            //subscribe for the event
            obj.MySampleEvent += objTest.onMyEventRaised1;
            obj.MySampleEvent += objTest2.onMyEventRaised2;
            //Calling Event Handler Method
            obj.RaiseMyEvent();


        }
    }
}
